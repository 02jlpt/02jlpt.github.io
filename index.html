}); // Hàm vẽ bụi (Được đơn giản hóa từ mã nguồn) function animateDust() { context.clearRect(0, 0, ParticleCanvas.width, ParticleCanvas.height); // Các logic vẽ particle thêm vào đây requestAnimationFrame(animateDust); iframe.setAttribute("frameborder", "0"); iframe.setAttribute("allowfullscreen", "1"); iframe.setAttribute("allow", "accelerometer; autoplay; encrypted-media; gyroscope; picture-in-picture"); div.parentNode.replaceChild(iframe, div);
